Function Total Bayar Pengunaan Listrik
   Declare Total Bayar, HargaPerkWh ,Pengunaan Listrik :Num
BEGIN
   Input(Pengunaan Listrik)
   If Pengunaan Listrik <= 100
        THEN
            HargaPerkWh <- 2000 rupiah
   ElseIF Pengunaan Listrik > 100  and Pengunaan Listrik <= 200 
        THEN
            HargaPerkWh <- 3000 rupiah
   Else 
        THEN
	        HargaPerkWh <- 5000 rupiah
   Endif
Total Bayar <- HargaPerkWh * Pengunaan Listrik 
End
